% ============================================================
% SLIDE 1: The Problem
% ============================================================
\begin{frame}{The Problem: Not All Tokens Are Equal in RLHF}

Standard PPO treats every token equally in the policy gradient:
\[
\mathcal{L}_{\text{PPO}} = \frac{1}{T}\sum_{t=1}^{T} \min\!\bigl(r_t \hat{A}_t,\; \text{clip}(r_t)\,\hat{A}_t\bigr)
\]

\textbf{But tokens contribute unequally to learning:}
\begin{itemize}
    \item \textit{``Let $x$ = \alert{3 + 4 = 7}. Therefore the answer is \alert{\textbackslash boxed\{7\}}''} --- the reasoning step matters
    \item \textit{``The answer is''} --- boilerplate, no learning signal
    \item Function words (``the'', ``is'', ``let'') are noise in the gradient
\end{itemize}

\vspace{0.3cm}
\textbf{Question:} Can we weight tokens by importance to get \textit{faster learning} and \textit{lower KL divergence}?

\vspace{0.3cm}
\textbf{TI-DPO} (Yang et al., 2025) showed this works for DPO. \textbf{Does it transfer to PPO?}
\end{frame}

% ============================================================
% SLIDE 2: Why This Is Hard
% ============================================================
\begin{frame}{Why This Is Harder Than It Looks}

\textbf{DPO vs PPO --- a fundamental difference:}

\begin{columns}
\begin{column}{0.48\textwidth}
\textbf{DPO (offline)}
\begin{itemize}
    \item Paired (chosen, rejected) sequences
    \item Single pass over data
    \item Bias doesn't accumulate
    \item Token importance = ``which tokens differ between chosen/rejected?''
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{PPO (online)}
\begin{itemize}
    \item Single sequence + scalar reward
    \item Hundreds of gradient steps
    \item \alert{Bias accumulates over training}
    \item Token importance = ??? (no contrastive signal)
\end{itemize}
\end{column}
\end{columns}

\vspace{0.5cm}
Importance weighting introduces a \alert{persistent bias} in the PPO gradient that compounds over training steps.

\end{frame}

% ============================================================
% SLIDE 3: TI-DPO Background
% ============================================================
\begin{frame}{Background: TI-DPO (Yang et al., 2025)}

Token-Importance guided DPO weights the preference loss per-token:
\[
\mathcal{L}_{\text{TI-DPO}} = -\log\sigma\!\Bigl(\beta\sum_t \alert{w(t)}\bigl[\log\pi(a_t^w|s_t) - \log\pi(a_t^\ell|s_t)\bigr]\Bigr)
\]

\textbf{Their importance scoring methods:}

\begin{table}
\small
\begin{tabular}{lll}
\toprule
Method & Signal & Cost \\
\midrule
Gradient & $\|\nabla_{\text{emb}}\|_1$ per token & 1 backward \\
Attention & Mean attention received & 1 forward \\
Hybrid & Gradient + Gaussian prior & 1 backward + 1 forward \\
TD Error & $|V(s_t) - V(s_{t+1}) - r_t|$ & Free \\
\bottomrule
\end{tabular}
\end{table}

Works well for DPO: +2--5\% on preference benchmarks.

\textbf{Our goal:} Adapt this to PPO for math reasoning (GSM8K, MATH).

\end{frame}

% ============================================================
% SLIDE 4: Why Entropy (not Advantage)
% ============================================================
\begin{frame}{Why Entropy, Not Advantage?}

\textbf{Advantage is meaningless for per-token importance in LLM PPO:}
\begin{itemize}
    \item Reward is a \textbf{single scalar} assigned at the last token
    \item GAE propagates it backwards: $\hat{A}_t = (0.95)^{T-t} \cdot R$
    \item This is just \textbf{positional decay} --- not a content signal
    \item Weighting by $|\hat{A}_t|$ = ``weight later tokens more'' $\to$ meaningless
\end{itemize}

\vspace{0.3cm}
\textbf{Entropy is a genuine content-based signal:}
\[
w(t) = \frac{H\bigl(\pi(\cdot | s_t)\bigr)}{\text{mean}(H)}, \qquad H(\pi) = -\sum_v \pi(v) \log \pi(v)
\]

\begin{itemize}
    \item \textbf{High entropy} $\to$ model is uncertain $\to$ multiple plausible continuations $\to$ \textcolor{good}{room to learn}
    \item \textbf{Low entropy} $\to$ model is confident $\to$ one dominant token $\to$ less to gain
    \item \textbf{Free:} computed from existing logits, zero extra cost
    \item \textbf{Content-aware:} varies based on what the token \textit{means}, not its position
\end{itemize}

\end{frame}

% SLIDE 5: Entropy Intuition
% ============================================================
\begin{frame}{Entropy Intuition: Where Should the Model Focus?}

\textbf{Example:} ``Solve: What is $15 \times 7$?''

\vspace{0.2cm}
\begin{center}
\begin{tabular}{lcl}
\toprule
Token & $H(\pi)$ & Interpretation \\
\midrule
``Let'' & 0.2 & Boilerplate --- model is sure \\
``me'' & 0.1 & Continuation is obvious \\
``calculate'' & 0.8 & \textbf{Multiple valid reasoning strategies} \\
``15'' & 0.3 & Copying from the problem \\
``$\times$'' & 0.1 & Determined by context \\
``7'' & 0.2 & Copying from the problem \\
``='' & \alert{1.4} & \textbf{Critical: model must compute the answer!} \\
``105'' & \alert{1.2} & \textbf{The actual answer --- high uncertainty} \\
``.'' & 0.1 & Punctuation \\
\bottomrule
\end{tabular}
\end{center}

\vspace{0.2cm}
Entropy naturally identifies \textbf{decision points} where the model's choice matters most for correctness. These are exactly the tokens where PPO's gradient signal is most valuable.
\end{frame}

% ============================================================
% SLIDE 6: The Degradation
% ============================================================
\begin{frame}{The Catch: Fixed-Intensity Entropy Weighting Degrades}

\textbf{At 150 episodes, the picture reverses:}

\begin{table}
\small
\begin{tabular}{lccc}
\toprule
Method & R(Q4) at ep 100 & R(Q4) at ep 150 & Trend \\
\midrule
PPO baseline & 0.791 & \textbf{0.850} & $\uparrow$ keeps improving \\
Entropy (fixed) & \textbf{0.872} & 0.797 & \textcolor{bad}{$\downarrow$ degraded} \\
\bottomrule
\end{tabular}
\end{table}

\vspace{0.3cm}
\textbf{Why?} Early in training, high-entropy tokens are genuinely uncertain decision points. But as the model learns:
\begin{itemize}
    \item Entropy landscape shifts --- high-entropy tokens become random noise, not meaningful uncertainty
    \item The biased gradient accumulates: over-updating some tokens, under-updating others
    \item Result: reward plateaus then declines
\end{itemize}

\vspace{0.2cm}
This is a \alert{fundamental property} of non-uniform weighting in online policy gradient, not a bug.
\end{frame}

% ============================================================
% SLIDE 7: The Math --- Bias-Variance Tradeoff
% ============================================================
\begin{frame}{The Key Insight: Bias-Variance Tradeoff}

The PPO gradient with importance weights:
\[
g_w = \mathbb{E}_t\bigl[w(t) \cdot \nabla\log\pi(a_t|s_t) \cdot \hat{A}_t\bigr]
\]

\textbf{Bias of this estimator:}
\[
\text{Bias} = \mathbb{E}[g_w] - \mathbb{E}[g] = \text{Cov}(w, f) \neq 0
\]
where $f = \nabla\log\pi \cdot \hat{A}$.

\vspace{0.3cm}

\begin{columns}
\begin{column}{0.48\textwidth}
\textbf{Early training:}
\[
\underbrace{\text{Var}(g)}_{\text{large}} \gg \underbrace{|\text{Bias}|}_{\text{small}}
\]
Entropy weighting reduces variance $\to$ \textcolor{good}{helps}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Late training:}
\[
\underbrace{|\text{Bias}|}_{\text{accumulated}} \gg \underbrace{\text{Var}(g)}_{\text{small}}
\]
Bias dominates $\to$ \textcolor{bad}{hurts}
\end{column}
\end{columns}

\vspace{0.3cm}
Explains why TI-DPO works (offline, no accumulation) but fixed TI-PPO fails (online, compounds).

\end{frame}

% ============================================================
% SLIDE 8: Our Solution --- AITI
% ============================================================
\begin{frame}{Our Solution: Adaptive Intensity Token Importance (AITI)}

\textbf{Decay the entropy-weighting intensity over training:}
\[
w_{\text{AITI}}(t) = 1 + \alert{\varepsilon(\text{step})} \cdot \bigl(H(t)_{\text{norm}} - 1\bigr)
\]

\begin{itemize}
    \item $H(t)_{\text{norm}}$ = normalized per-token entropy (mean $\approx 1$)
    \item $\varepsilon$ decays from 1 $\to$ 0 (or to a residual minimum)
    \item $\varepsilon = 1$: full entropy weighting (maximum variance reduction)
    \item $\varepsilon = 0$: uniform PPO (zero bias)
\end{itemize}

\vspace{0.3cm}
\textbf{Analogy:} Learning rate scheduling controls \textit{step size}. AITI controls \textit{gradient quality}.

\begin{table}
\small
\begin{tabular}{lcc}
\toprule
Phase & Learning Rate & AITI Intensity ($\varepsilon$) \\
\midrule
Early & High (explore) & High (focus on uncertain tokens) \\
Late & Low (fine-tune) & Low (unbiased gradients for all tokens) \\
\bottomrule
\end{tabular}
\end{table}

\end{frame}

% ============================================================
% SLIDE 9: MOAI
% ============================================================
\begin{frame}{MOAI: MSE-Optimal Adaptive Intensity}

Instead of a hand-tuned decay schedule, \textbf{compute optimal $\varepsilon^*$ per step}:

\[
\varepsilon^* = \arg\min_\varepsilon\; \text{MSE}(g_\varepsilon) = \arg\min_\varepsilon\; \bigl[\text{Bias}(\varepsilon)^2 + \text{Var}(\varepsilon)\bigr]
\]

\textbf{Closed-form solution} (with monotone constraint $\varepsilon^* \leq \varepsilon^*_{\text{prev}}$):

\[
\varepsilon^* = \text{clip}\!\left(\frac{-C}{2\rho},\; 0,\; \varepsilon_{\text{prev}}\right)
\]

where $C = \text{Cov}(H_{\text{norm}}, f)$ and $\rho = \text{Var}(H_{\text{norm}}) \cdot \mathbb{E}[f^2]$, estimated via EMA.

\vspace{0.3cm}
\textbf{Key discovery:} MOAI converges to $\varepsilon^* \approx 0.3$--$0.5$, \textit{not} zero.

$\Rightarrow$ The optimal strategy is \textbf{permanent partial entropy weighting}, not full decay.

The model always benefits from some focus on uncertain tokens.

\end{frame}

% ============================================================
% SLIDE 14: Takeaways
% ============================================================
\begin{frame}{Key Takeaways}

\begin{enumerate}
    \item \textbf{Entropy is the right importance signal for PPO}
    \begin{itemize}
        \item Advantage is positional (not content-based) with scalar rewards
        \item Entropy identifies \textit{decision points} where the model is uncertain
        \item Free to compute from existing logits
    \end{itemize}

    \vspace{0.2cm}
    \item \textbf{Fixed entropy weighting degrades due to bias accumulation}
    \begin{itemize}
        \item $\text{Bias} = \text{Cov}(w, f) \neq 0$, compounds over gradient steps
        \item Helps early (variance reduction), hurts late (bias dominates)
    \end{itemize}

    \vspace{0.2cm}
    \item \textbf{AITI/MOAI solve this with adaptive intensity decay}
    \begin{itemize}
        \item AITI: scheduled decay $\varepsilon: 1 \to 0$
        \item MOAI: closed-form optimal $\varepsilon^*$ per step, converges to $\sim$0.3--0.5
    \end{itemize}

    \vspace{0.2cm}
    \item \textbf{Benefit is strongest in high-variance regimes}
    \begin{itemize}
        \item Small batch sizes, sparse rewards, early training, harder tasks
        \item A variance reduction technique --- complementary to, not replacing, PPO
    \end{itemize}
\end{enumerate}

\end{frame}
